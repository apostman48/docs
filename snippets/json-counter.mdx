import React, { useState } from 'react';

function JsonItemCounter({ data, title = "JSON Structure" }) {
  const [collapsed, setCollapsed] = useState({});

  const toggleCollapse = (path) => {
    setCollapsed(prev => ({
      ...prev,
      [path]: !prev[path]
    }));
  };

  const countItems = (obj) => {
    if (Array.isArray(obj)) {
      return obj.length;
    }
    if (obj && typeof obj === 'object') {
      return Object.keys(obj).length;
    }
    return 0;
  };

  const renderValue = (value, key, path = '', level = 0) => {
    const currentPath = path ? `${path}.${key}` : key;
    const indent = level * 20;
    
    if (Array.isArray(value)) {
      const itemCount = value.length;
      const isCollapsed = collapsed[currentPath];
      
      return (
        <div key={currentPath} style={{ marginLeft: `${indent}px` }}>
          <div 
            onClick={() => toggleCollapse(currentPath)}
            style={{ 
              cursor: 'pointer', 
              display: 'flex', 
              alignItems: 'center',
              padding: '6px 12px',
              borderRadius: '6px',
              backgroundColor: 'rgba(0, 0, 0, 0.05)',
              marginBottom: '4px',
              border: '1px solid rgba(0, 0, 0, 0.1)',
              transition: 'all 0.2s ease'
            }}
            onMouseEnter={(e) => {
              e.target.style.backgroundColor = 'rgba(0, 0, 0, 0.08)';
            }}
            onMouseLeave={(e) => {
              e.target.style.backgroundColor = 'rgba(0, 0, 0, 0.05)';
            }}
          >
            <span style={{ 
              marginRight: '8px',
              transform: isCollapsed ? 'rotate(-90deg)' : 'rotate(0deg)',
              transition: 'transform 0.2s ease',
              fontSize: '12px'
            }}>
              ▼
            </span>
            <span style={{ 
              color: '#2563eb',
              fontWeight: '600',
              fontFamily: 'monospace'
            }}>
              "{key}"
            </span>
            <span style={{ margin: '0 8px', color: '#6b7280' }}>:</span>
            <span style={{ color: '#6b7280' }}>[</span>
            <span style={{ 
              backgroundColor: '#f59e0b',
              color: 'white',
              padding: '2px 8px',
              borderRadius: '12px',
              fontSize: '11px',
              margin: '0 8px',
              fontWeight: '600'
            }}>
              {itemCount} items
            </span>
            <span style={{ color: '#6b7280' }}>]</span>
          </div>
          
          {!isCollapsed && (
            <div style={{ 
              borderLeft: '2px solid #e5e7eb',
              marginLeft: '10px',
              paddingLeft: '10px',
              marginTop: '4px'
            }}>
              {value.map((item, index) => (
                <div key={index}>
                  {renderValue(item, index, currentPath, level + 1)}
                </div>
              ))}
            </div>
          )}
        </div>
      );
    }
    
    if (value && typeof value === 'object') {
      const itemCount = Object.keys(value).length;
      const isCollapsed = collapsed[currentPath];
      
      return (
        <div key={currentPath} style={{ marginLeft: `${indent}px` }}>
          <div 
            onClick={() => toggleCollapse(currentPath)}
            style={{ 
              cursor: 'pointer', 
              display: 'flex', 
              alignItems: 'center',
              padding: '6px 12px',
              borderRadius: '6px',
              backgroundColor: 'rgba(0, 0, 0, 0.05)',
              marginBottom: '4px',
              border: '1px solid rgba(0, 0, 0, 0.1)',
              transition: 'all 0.2s ease'
            }}
            onMouseEnter={(e) => {
              e.target.style.backgroundColor = 'rgba(0, 0, 0, 0.08)';
            }}
            onMouseLeave={(e) => {
              e.target.style.backgroundColor = 'rgba(0, 0, 0, 0.05)';
            }}
          >
            <span style={{ 
              marginRight: '8px',
              transform: isCollapsed ? 'rotate(-90deg)' : 'rotate(0deg)',
              transition: 'transform 0.2s ease',
              fontSize: '12px'
            }}>
              ▼
            </span>
            <span style={{ 
              color: '#2563eb',
              fontWeight: '600',
              fontFamily: 'monospace'
            }}>
              "{key}"
            </span>
            <span style={{ margin: '0 8px', color: '#6b7280' }}>:</span>
            <span style={{ color: '#6b7280' }}>{"{"}</span>
            <span style={{ 
              backgroundColor: '#10b981',
              color: 'white',
              padding: '2px 8px',
              borderRadius: '12px',
              fontSize: '11px',
              margin: '0 8px',
              fontWeight: '600'
            }}>
              {itemCount} properties
            </span>
            <span style={{ color: '#6b7280' }}>{"}"}</span>
          </div>
          
          {!isCollapsed && (
            <div style={{ 
              borderLeft: '2px solid #e5e7eb',
              marginLeft: '10px',
              paddingLeft: '10px',
              marginTop: '4px'
            }}>
              {Object.entries(value).map(([k, v]) => (
                renderValue(v, k, currentPath, level + 1)
              ))}
            </div>
          )}
        </div>
      );
    }
    
    // Primitive values
    const getValueColor = (val) => {
      if (typeof val === 'string') return '#10b981';
      if (typeof val === 'number') return '#f59e0b';
      if (typeof val === 'boolean') return '#8b5cf6';
      return '#6b7280';
    };
    
    const formatValue = (val) => {
      if (typeof val === 'string') return `"${val}"`;
      return String(val);
    };
    
    return (
      <div key={currentPath} style={{ 
        marginLeft: `${indent}px`,
        padding: '3px 12px',
        marginBottom: '2px',
        display: 'flex',
        alignItems: 'center'
      }}>
        <span style={{ 
          color: '#2563eb',
          fontWeight: '600',
          fontFamily: 'monospace'
        }}>
          "{key}"
        </span>
        <span style={{ margin: '0 8px', color: '#6b7280' }}>:</span>
        <span style={{ 
          color: getValueColor(value),
          fontFamily: 'monospace'
        }}>
          {formatValue(value)}
        </span>
      </div>
    );
  };

  return (
    <div style={{ 
      border: '1px solid #e5e7eb',
      borderRadius: '8px',
      padding: '16px',
      backgroundColor: '#fafafa',
      fontFamily: 'monospace',
      fontSize: '14px',
      lineHeight: '1.5',
      margin: '16px 0'
    }}>
      <h4 style={{ 
        margin: '0 0 16px 0',
        color: '#1f2937',
        fontFamily: 'system-ui, sans-serif',
        fontSize: '18px',
        fontWeight: '600'
      }}>
        {title}
      </h4>
      
      <div>
        {typeof data === 'object' && data !== null ? (
          Object.entries(data).map(([key, value]) => renderValue(value, key, '', 0))
        ) : (
          <div style={{ color: '#ef4444', fontWeight: '500' }}>
            Invalid JSON data provided
          </div>
        )}
      </div>
    </div>
  );
}
