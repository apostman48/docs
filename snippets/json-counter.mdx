function JsonItemCounter({ data, title = "JSON Structure" }) {
  if (!data || typeof data !== 'object') {
    return (
      <div style={{
        border: '1px solid #ef4444',
        borderRadius: '8px',
        padding: '16px',
        backgroundColor: '#fef2f2',
        color: '#dc2626',
        margin: '16px 0',
        fontWeight: '500'
      }}>
        <strong>Error:</strong> Invalid JSON data provided
      </div>
    );
  }

  const countItems = (obj) => {
    if (Array.isArray(obj)) return obj.length;
    if (obj && typeof obj === 'object') return Object.keys(obj).length;
    return 0;
  };

  const renderJsonTree = (obj, depth = 0) => {
    const indent = depth * 20;
    
    return Object.entries(obj).map(([key, value], index) => {
      const isArray = Array.isArray(value);
      const isObject = value && typeof value === 'object' && !isArray;
      const isPrimitive = !isArray && !isObject;
      
      return (
        <div key={`${key}-${index}`} style={{ marginLeft: `${indent}px`, marginBottom: '8px' }}>
          {isPrimitive ? (
            // Primitive value display
            <div style={{
              display: 'flex',
              alignItems: 'center',
              padding: '6px 12px',
              fontFamily: 'ui-monospace, SFMono-Regular, Monaco, Consolas, monospace',
              fontSize: '13px',
              backgroundColor: '#f9fafb',
              borderRadius: '4px'
            }}>
              <span style={{ color: '#1e40af', fontWeight: '600' }}>"{key}"</span>
              <span style={{ margin: '0 8px', color: '#6b7280' }}>:</span>
              <span style={{ 
                color: typeof value === 'string' ? '#059669' : 
                       typeof value === 'number' ? '#dc2626' : 
                       typeof value === 'boolean' ? '#7c3aed' : '#6b7280'
              }}>
                {typeof value === 'string' ? `"${value}"` : String(value)}
              </span>
            </div>
          ) : (
            // Array or Object container
            <div>
              <div style={{
                display: 'flex',
                alignItems: 'center',
                padding: '8px 12px',
                backgroundColor: '#f1f5f9',
                border: '1px solid #cbd5e1',
                borderRadius: '6px',
                marginBottom: '6px',
                fontFamily: 'ui-monospace, SFMono-Regular, Monaco, Consolas, monospace',
                fontSize: '13px',
                cursor: 'default'
              }}>
                <span style={{ color: '#1e40af', fontWeight: '700' }}>"{key}"</span>
                <span style={{ margin: '0 8px', color: '#6b7280' }}>:</span>
                <span style={{ color: '#6b7280', fontSize: '14px' }}>{isArray ? '[' : '{'}</span>
                <span style={{
                  backgroundColor: isArray ? '#f59e0b' : '#10b981',
                  color: 'white',
                  padding: '3px 8px',
                  borderRadius: '12px',
                  fontSize: '11px',
                  margin: '0 8px',
                  fontWeight: '600',
                  minWidth: 'fit-content'
                }}>
                  {countItems(value)} {isArray ? 'items' : 'props'}
                </span>
                <span style={{ color: '#6b7280', fontSize: '14px' }}>{isArray ? ']' : '}'}</span>
              </div>
              
              {/* Nested content */}
              <div style={{
                borderLeft: '2px solid #e2e8f0',
                marginLeft: '12px',
                paddingLeft: '12px',
                paddingTop: '4px'
              }}>
                {renderJsonTree(value, depth + 1)}
              </div>
            </div>
          )}
        </div>
      );
    });
  };

  return (
    <div style={{
      border: '1px solid #e2e8f0',
      borderRadius: '8px',
      padding: '20px',
      backgroundColor: '#ffffff',
      margin: '20px 0',
      boxShadow: '0 1px 3px 0 rgba(0, 0, 0, 0.1), 0 1px 2px 0 rgba(0, 0, 0, 0.06)'
    }}>
      <h4 style={{
        margin: '0 0 16px 0',
        color: '#111827',
        fontSize: '18px',
        fontWeight: '600',
        fontFamily: 'system-ui, -apple-system, BlinkMacSystemFont, sans-serif'
      }}>
        {title}
      </h4>
      
      <div style={{ overflow: 'auto' }}>
        {renderJsonTree(data)}
      </div>
    </div>
  );
}
